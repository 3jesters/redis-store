Rack cache

Provides a Redis store for HTTP caching.

Rack application
require "rack"
require "rack/cache"
require "redis-store"
require "application"
use Rack::Cache,
  :metastore   => 'redis://localhost:6379/0/metastore',
  :entitystore => 'redis://localhost:6380/0/entitystore'
run Application.new
